## WHAT THIS IS

A self-contained HTML contact form snippet built by Localhost:314
(https://localhost314.com). Drop the block into any HTML page and it works.
Submissions POST to the Localhost:314 API which forwards emails via Resend
to tristanarbet@gmail.com.

CORS is enabled — the block works on any domain or subdomain, no server
setup required on the embedding site.


## API ENDPOINT

URL:           https://localhost314.com/api/contact
Method:        POST
Content-Type:  application/json
CORS:          Allowed from all origins (*)

Request body (JSON):
  {
    "name":     "string  — required",
    "email":    "string  — required",
    "message":  "string  — required",
    "platform": "string  — optional, appended to email subject"
  }

Success response 200:
  { "success": true }

Error response 500:
  { "error": "string", "code": number }
  (code is the upstream HTTP status, e.g. 403, 422, 500)


## CONFIG VARIABLES

Place these in a <script> block ABOVE the pasted block HTML.
All four are optional — the block has safe defaults for each.

  var L314_WATERMARK = false;
  // true  → shows small "Powered by Localhost:314" link at the bottom
  // false → hides it
  // Rule: false on any Localhost:314-owned site, true everywhere else

  var L314_API = 'https://localhost314.com/api/contact';
  // The API endpoint. Do not change unless the API moves.

  var L314_PLATFORM = '';
  // Name of the site/platform embedding the form.
  // Empty string = no suffix.
  // e.g. 'Dispatchy'  →  subject: "New message from John on Dispatchy"
  // e.g. 'Globe'      →  subject: "New message from Jane on Globe"

  var L314_ERROR_CONTACT = 'Reach us directly at contact@localhost314.com';
  // The tail of every error message shown to the user.
  // Format on screen: "Error (403) — [this value]"
  // Examples:
  //   'Reach us directly at contact@localhost314.com'
  //   'Email us at support@dispatchy.com'
  //   'Call us at 555-123-4567'
  //   'Visit our help center at help.example.com'

<script>
(function(){
  if(typeof L314_WATERMARK==='undefined')L314_WATERMARK=false;
  if(typeof L314_API==='undefined')L314_API='https://localhost314.com/api/contact';
  if(typeof L314_PLATFORM==='undefined')L314_PLATFORM='';
  if(typeof L314_ERROR_CONTACT==='undefined')L314_ERROR_CONTACT='Reach us directly at contact@localhost314.com';
  if(L314_WATERMARK)document.getElementById('l314cf-wm').style.display='block';
  var form=document.getElementById('l314cf-form');
  var btn=document.getElementById('l314cf-btn');
  var stat=document.getElementById('l314cf-status');
  form.addEventListener('submit',function(e){
    e.preventDefault();
    var name=document.getElementById('l314cf-name').value.trim();
    var email=document.getElementById('l314cf-email').value.trim();
    var message=document.getElementById('l314cf-msg').value.trim();
    if(!name||!email||!message){stat.className='l314cf-status err';stat.textContent='Please fill in all fields.';return;}
    btn.disabled=true;btn.textContent='Sending…';stat.className='l314cf-status';stat.textContent='';
    var payload={name:name,email:email,message:message};
    if(L314_PLATFORM)payload.platform=L314_PLATFORM;
    fetch(L314_API,{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify(payload)})
      .then(function(r){return r.json().then(function(j){return{ok:r.ok,body:j};});})
      .then(function(r){
        if(r.ok){stat.className='l314cf-status ok';stat.textContent="Message sent! We'll get back to you soon.";form.reset();}
        else{var code=r.body&&r.body.code?r.body.code:'';stat.className='l314cf-status err';stat.textContent=code?('Error ('+code+') — '+L314_ERROR_CONTACT):(L314_ERROR_CONTACT||'Something went wrong.');}
      })
      .catch(function(){stat.className='l314cf-status err';stat.textContent='Could not reach the server — '+L314_ERROR_CONTACT;})
      .finally(function(){btn.disabled=false;btn.textContent='Send Message';});
  });
}());
</script>
